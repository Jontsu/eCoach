30      public final void tarkistaSuoritusStatus() {
31  
32          if (pelaaja.getHarjoitusLista().getLista().get(valitunHarjoituksenIndeksi).getSuoritus().suoritusStatus() == true) {
33              suoritusNappi.setEnabled(false);
34              linkkiLabel.setText(pelaaja.getHarjoitusLista().getLista().get(valitunHarjoituksenIndeksi).getSuoritus().getSuoritusLinkki());
35              linkkiLabel.setEditable(false);
36          } else {
37              suoritusNappi.setEnabled(true);
38              linkkiLabel.setText("Lisää tähän suoritus nauhoituksen linkki");
39              linkkiLabel.setEditable(true);
40          }
41      }

43      public final void paivitaKuvaus() {
44  
45          String apuString;

47          harjoituksenKuvaus.setText(apuString);
48          harjoituksenKuvaus.setEditable(false);
49      }
